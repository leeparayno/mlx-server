### Task 4: Memory Leak Testing Tool (2-3 hours)

**Goal:** Implement automated memory leak detection through 1000-request cycles.

**Deliverables:**
- Memory monitoring script
- 1000-request test harness
- Memory leak report

**Implementation:**

#### Step 1: Create Memory Monitor Script
```bash
# scripts/memory_test.sh
#!/bin/bash
# Memory leak testing for MLX Server

ITERATIONS=1000
MODEL="mlx-community/Qwen2.5-0.5B-Instruct-4bit"
PORT=8080
RESULTS_DIR="./test-results/memory-leak-$(date +%Y%m%d-%H%M%S)"

mkdir -p "$RESULTS_DIR"

# Start server
mlx-server --model "$MODEL" --port $PORT &
SERVER_PID=$!
sleep 30  # Warmup

# Record baseline memory
BASELINE=$(ps -o rss= -p $SERVER_PID)
echo "Baseline memory: ${BASELINE}KB" | tee "$RESULTS_DIR/memory.log"

# Run request cycles
for i in $(seq 1 $ITERATIONS); do
    # Send request
    curl -s -X POST http://localhost:$PORT/v1/completions \
        -H "Content-Type: application/json" \
        -d "{\"model\":\"test\",\"prompt\":\"Test $i\",\"max_tokens\":10,\"stream\":false}" \
        > /dev/null

    # Sample memory every 100 requests
    if [ $((i % 100)) -eq 0 ]; then
        CURRENT=$(ps -o rss= -p $SERVER_PID)
        DELTA=$((CURRENT - BASELINE))
        echo "[$i] Memory: ${CURRENT}KB (Δ${DELTA}KB)" | tee -a "$RESULTS_DIR/memory.log"
    fi
done

# Final memory check
FINAL=$(ps -o rss= -p $SERVER_PID)
TOTAL_DELTA=$((FINAL - BASELINE))
LEAK_RATE=$(echo "scale=2; $TOTAL_DELTA / $ITERATIONS" | bc)

echo "Final memory: ${FINAL}KB" | tee -a "$RESULTS_DIR/memory.log"
echo "Total delta: ${TOTAL_DELTA}KB" | tee -a "$RESULTS_DIR/memory.log"
echo "Leak rate: ${LEAK_RATE}KB/request" | tee -a "$RESULTS_DIR/memory.log"

# Cleanup
kill $SERVER_PID

# Verdict
if [ "$TOTAL_DELTA" -lt 10240 ]; then  # <10MB total growth
    echo "✅ PASS: No memory leak detected" | tee -a "$RESULTS_DIR/memory.log"
    exit 0
else
    echo "❌ FAIL: Memory leak detected (${TOTAL_DELTA}KB)" | tee -a "$RESULTS_DIR/memory.log"
    exit 1
fi
```

```bash
# scripts/load_test_enhanced.sh
#!/bin/bash
set -e

CONCURRENCY=${1:-16}
NUM_REQUESTS=${2:-100}
MAX_TOKENS=${3:-50}
MODEL="mlx-community/Qwen2.5-0.5B-Instruct-4bit"
PORT=8080
RESULTS_DIR="./test-results/load-test-c${CONCURRENCY}-$(date +%Y%m%d-%H%M%S)"

mkdir -p "$RESULTS_DIR"

echo "Load Test Configuration:"
echo "  Concurrency: $CONCURRENCY"
echo "  Requests: $NUM_REQUESTS"
echo "  Max Tokens: $MAX_TOKENS"
echo "  Results: $RESULTS_DIR"

# Start server
mlx-server --model "$MODEL" --port $PORT &
SERVER_PID=$!
trap "kill $SERVER_PID 2>/dev/null || true" EXIT

# Wait for server
for i in {1..60}; do
    if curl -s http://localhost:$PORT/health > /dev/null 2>&1; then
        echo "Server ready!"
        break
    fi
    sleep 1
done

# Monitor memory in background
(
    while true; do
        if ps -p $SERVER_PID > /dev/null 2>&1; then
            RSS=$(ps -o rss= -p $SERVER_PID)
            echo "$(date +%s),$RSS" >> "$RESULTS_DIR/memory.csv"
        fi
        sleep 1
    done
) &
MONITOR_PID=$!

# Run load test with detailed metrics
echo "Starting load test..."
START_TIME=$(date +%s)

seq 1 $NUM_REQUESTS | xargs -P $CONCURRENCY -I {} sh -c "
    REQ_START=\$(date +%s%3N)
    RESPONSE=\$(curl -s -w '\n%{time_total}' -X POST http://localhost:$PORT/v1/completions \
        -H 'Content-Type: application/json' \
        -d '{\"model\":\"test\",\"prompt\":\"Request {}\",\"max_tokens\":$MAX_TOKENS,\"stream\":false}')
    REQ_TIME=\$(echo \"\$RESPONSE\" | tail -1)
    REQ_END=\$(date +%s%3N)
    LATENCY=\$((REQ_END - REQ_START))
    echo \"{},\$LATENCY,\$REQ_TIME\" >> '$RESULTS_DIR/latencies.csv'
"

END_TIME=$(date +%s)
DURATION=$((END_TIME - START_TIME))

# Stop monitoring
kill $MONITOR_PID 2>/dev/null || true

# Get final metrics
METRICS=$(curl -s http://localhost:$PORT/metrics)

# Calculate statistics
echo "timestamp,latency_ms,req_time_s" > "$RESULTS_DIR/latencies_header.csv"
cat "$RESULTS_DIR/latencies.csv" | sort -t, -k2 -n > "$RESULTS_DIR/latencies_sorted.csv"

# Generate report
cat > "$RESULTS_DIR/report.md" << EOF
# Load Test Report

**Date:** $(date)
**Concurrency:** $CONCURRENCY
**Total Requests:** $NUM_REQUESTS
**Max Tokens:** $MAX_TOKENS

## Results

- **Duration:** ${DURATION}s
- **Throughput:** $(echo "scale=2; $NUM_REQUESTS / $DURATION" | bc) req/s
- **Avg Latency:** $(awk -F, '{sum+=$2; count++} END {print sum/count}' "$RESULTS_DIR/latencies.csv") ms

## Latency Percentiles

$(python3 -c "
import sys
data = [float(line.split(',')[1]) for line in open('$RESULTS_DIR/latencies_sorted.csv')]
print(f'- p50: {data[len(data)//2]:.2f}ms')
print(f'- p95: {data[int(len(data)*0.95)]:.2f}ms')
print(f'- p99: {data[int(len(data)*0.99)]:.2f}ms')
")

## Memory Usage

$(python3 -c "
import sys
data = [int(line.split(',')[1]) for line in open('$RESULTS_DIR/memory.csv')]
print(f'- Min: {min(data)/1024:.1f}MB')
print(f'- Max: {max(data)/1024:.1f}MB')
print(f'- Avg: {sum(data)/len(data)/1024:.1f}MB')
")

## Server Metrics

\`\`\`json
$METRICS
\`\`\`
EOF

cat "$RESULTS_DIR/report.md"
```

// benchmarks/Benchmark.swift
import Foundation
import ArgumentParser
import Core
import Scheduler
import Memory

@main
struct BenchmarkCommand: AsyncParsableCommand {
    static let configuration = CommandConfiguration(
        commandName: "mlx-benchmark",
        abstract: "Comprehensive benchmarking tool for MLX Server",
        version: "1.0.0"
    )

    @Option(name: .shortAndLong, help: "Model path")
    var model: String = "mlx-community/Qwen2.5-0.5B-Instruct-4bit"

    @Option(name: .long, help: "Number of warmup iterations")
    var warmup: Int = 10

    @Option(name: .long, help: "Number of benchmark iterations")
    var iterations: Int = 100

    @Option(name: .long, help: "Concurrent requests")
    var concurrency: Int = 16

    @Option(name: .long, help: "Max tokens per request")
    var maxTokens: Int = 50

    @Option(name: .long, help: "Output format (json|csv|markdown)")
    var format: String = "markdown"

    @Flag(name: .long, help: "Run all benchmark scenarios")
    var all: Bool = false

    func run() async throws {
        print("🔬 MLX Server Benchmark")
        print("Model: \(model)")
        print("Iterations: \(iterations) (warmup: \(warmup))")
        print("Concurrency: \(concurrency)")
        print()

        // Load model
        print("Loading model...")
        let engine = InferenceEngine()
        try await engine.initialize(modelPath: model)
        print("✅ Model loaded")

        // Create scheduler and batcher
        let scheduler = RequestScheduler()
        let batcher = ContinuousBatcher(
            scheduler: scheduler,
            engine: engine,
            config: ContinuousBatcher.Config(
                maxBatchSize: concurrency,
                eosTokenId: 2
            )
        )

        // Start batcher
        let batcherTask = Task {
            await batcher.start()
        }

        var results = BenchmarkResults()

        // Scenario 1: Single request latency
        print("\n📊 Scenario 1: Single Request Latency")
        let singleLatency = try await measureSingleRequestLatency(
            scheduler: scheduler,
            iterations: iterations
        )
        results.singleRequestLatency = singleLatency
        print("  Latency: \(String(format: "%.2f", singleLatency.mean))ms ±\(String(format: "%.2f", singleLatency.stddev))ms")

        // Scenario 2: Concurrent throughput
        print("\n📊 Scenario 2: Concurrent Throughput")
        let throughput = try await measureConcurrentThroughput(
            scheduler: scheduler,
            concurrency: concurrency,
            iterations: iterations
        )
        results.concurrentThroughput = throughput
        print("  Throughput: \(String(format: "%.2f", throughput.tokensPerSecond)) tok/s")
        print("  Requests/s: \(String(format: "%.2f", throughput.requestsPerSecond))")

        // Scenario 3: Memory efficiency
        print("\n📊 Scenario 3: Memory Efficiency")
        let memory = await measureMemoryEfficiency(batcher: batcher)
        results.memoryEfficiency = memory
        print("  KV Cache: \(String(format: "%.1f", memory.kvCacheMemoryMB))MB")
        print("  Total: \(String(format: "%.1f", memory.totalMemoryMB))MB")

        // Scenario 4: Batch size scaling
        if all {
            print("\n📊 Scenario 4: Batch Size Scaling")
            let scaling = try await measureBatchSizeScaling(
                scheduler: scheduler,
                engine: engine,
                sizes: [1, 4, 8, 16, 32]
            )
            results.batchSizeScaling = scaling
            for (size, perf) in scaling {
                print("  Batch \(size): \(String(format: "%.2f", perf)) tok/s")
            }
        }

        // Stop batcher
        await batcher.stop()
        batcherTask.cancel()

        // Output results
        print("\n" + "=".repeating(count: 60))
        print("Benchmark Complete!")
        print("=".repeating(count: 60))

        switch format {
        case "json":
            try outputJSON(results)
        case "csv":
            try outputCSV(results)
        default:
            outputMarkdown(results)
        }
    }

    // ... measurement functions ...
}

    iterations: Int

    var latencies: [Double] = []

    for i in 0..<iterations {
        let start = Date()

            prompt: "Benchmark test \(i)",
            maxTokens: 10
        )
        let (_, stream) = await scheduler.submit(request, priority: .normal)

        // Consume stream
        for try await _ in stream {}

        let latency = Date().timeIntervalSince(start) * 1000  // ms
        latencies.append(latency)
    }

    return calculateStats(latencies)
}

func measureConcurrentThroughput(
    scheduler: RequestScheduler,
    concurrency: Int,
    iterations: Int
) async throws -> ThroughputStats {
    let start = Date()
    var totalTokens = 0

    await withTaskGroup(of: Int.self) { group in
        for i in 0..<iterations {
            group.addTask {
                let request = InferenceRequest(
                    prompt: "Concurrent test \(i)",
                    maxTokens: 50
                )
                let (_, stream) = await scheduler.submit(request, priority: .normal)

                var tokenCount = 0
                for try await _ in stream {
                    tokenCount += 1
                }
                return tokenCount
            }
        }

        for await tokens in group {
            totalTokens += tokens
        }
    }

    let duration = Date().timeIntervalSince(start)
    return ThroughputStats(
        tokensPerSecond: Double(totalTokens) / duration,
        requestsPerSecond: Double(iterations) / duration,
        averageLatency: duration / Double(iterations) * 1000
    )
}
